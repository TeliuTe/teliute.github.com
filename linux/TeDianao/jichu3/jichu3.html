批量替换所有html文件，当前文件夹及子文件夹

#!/bin/bash
for i in $(find . -name '*.html' -print)
do#分号作为分隔符，-i是替换文件，$i是变量，文件名，由于后面替换的两个分号之间为空，所以是删除字符
	sed -i 's;<script src="fy.js"></script>;;g' $i
done
#HTML里空格是&nbsp; 没有Tab的代表字符，不过你可以用<pre></pre>把内容包起来，从而保留内容里面的Tab
